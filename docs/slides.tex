\section{Problem \& Solution}
\begin{frame}{The Challenge}
    \begin{itemize}
        \item \textbf{Problem:} Maintaining healthy eating habits is difficult.
        \item \textbf{Pain Points:}
        \begin{itemize}
            \item Manual calorie counting is tedious.
            \item Lack of personalized guidance.
            \item Nutritional data is hard to interpret.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Our Solution: AI Mindful Eating Agent}
    An intelligent conversational agent that:
    \begin{itemize}
        \item \textbf{Understands Natural Language:} "I had a burger" $\rightarrow$ Parsed Data.
        \item \textbf{Calculates Nutrition:} Calories, Protein, Carbs, Fat.
        \item \textbf{Learns Patterns:} "You've been eating a lot of sugar lately."
        \item \textbf{Provides Recommendations:} Context-aware advice.
    \end{itemize}
\end{frame}

\section{Key Features}
\begin{frame}{Key Features}
    \begin{enumerate}
        \item \textbf{Natural Language Processing:} Fuzzy matching for food items.
        \item \textbf{Supervisor Integration:} REST API for external control.
        \item \textbf{Memory Management:}
        \begin{itemize}
            \item Short-term: Session-based context.
            \item Long-term: MongoDB user history.
        \end{itemize}
        \item \textbf{Scalability:} Modular design allows adding new workers easily.
    \end{enumerate}
\end{frame}
